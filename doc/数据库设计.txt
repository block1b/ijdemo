如上7个表就是数据库中的表，全部使用string类型，数字在Java中转换计算。

DROP TABLE IF EXISTS class;
CREATE TABLE class (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY,
  classname varchar(45) unique NOT NULL,
  PRIMARY KEY (id)
) ;

DROP TABLE IF EXISTS course;
CREATE TABLE course (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY,
  courseid varchar(45)  unique NOT NULL,
  coursename varchar(45) NOT NULL,
  PRIMARY KEY (id)
) ;

DROP TABLE IF EXISTS grade;
CREATE TABLE grade (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY,
  serveid varchar(45) NOT NULL,
  userid varchar(45) NOT NULL,
  topicid varchar(45) NOT NULL,
  gd1 varchar(45) DEFAULT '0',
  gd2 varchar(45) DEFAULT '0',
  gd3 varchar(45) DEFAULT '0',
  gd4 varchar(45) DEFAULT '0',
  gdsum varchar(45) DEFAULT '0',
  comment varchar(45) DEFAULT '无',
  classrank varchar(45) DEFAULT '0',
  majorrank varchar(45) DEFAULT '0',
  PRIMARY KEY (id)
) ;

DROP TABLE IF EXISTS role;
CREATE TABLE role (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY,
  roleid varchar(45)  unique NOT NULL,
  rolename varchar(45)  unique NOT NULL,
  permissions varchar(45) DEFAULT 'r',
  PRIMARY KEY (id)
) ;


DROP TABLE IF EXISTS serve;
CREATE TABLE serve (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY,
  userid varchar(45) NOT NULL,
  courseid varchar(45) NOT NULL,
  PRIMARY KEY (id)
) ;

DROP TABLE IF EXISTS topic;
CREATE TABLE topic (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY,
  serviceid varchar(45) NOT NULL,
  topicid varchar(45) NOT NULL,
  topicname varchar(45) NOT NULL,
  PRIMARY KEY (id)
) ;

DROP TABLE IF EXISTS user;
CREATE TABLE user (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY,
  userid varchar(45)  unique NOT NULL,
  username varchar(45) NOT NULL,
  password varchar(45) DEFAULT '123456',
  classname varchar(45) DEFAULT 'null',
  roleid varchar(45) DEFAULT '4',
  PRIMARY KEY (id)
) ;
